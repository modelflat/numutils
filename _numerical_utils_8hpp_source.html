262 template <typename T>
263 std::vector<T> eliminate(Surface<T>&& A) {
264  size_t N = A.rowCount();
265 
266  for (size_t i = 0; i < N; ++i) {
267  // Search for maximum in this column
268  T maxEl = std::abs( A.at(i, i) );
269  size_t maxRow = i;
270  for (size_t k = i + 1; k < N; ++k) {
271  const T el = std::abs(A.at(k, i));
272  if (el > maxEl) {
273  maxEl = el; maxRow = k;
274  }
275  }
276 
277  if (maxRow != i) {
278  // Swap maximum row with current row (column by column)
279  for (size_t k = i; k < N + 1; ++k) {
280  std::swap(A.at(maxRow, k), A.at(i, k));
281  }
282  }
283 
284  // eliminate rows below
285  for (size_t k = i + 1; k < N; ++k) {
286  T c = -A.at(k, i) / A.at(i, i);
287  for (size_t j = i; j < N + 1; ++j) {
288  if (i == j) {
289  A.at(k, j) = 0;
290  } else {
291  A.at(k, j) += c * A.at(i, j);
292  }
293  }
294  }
295  }
296 
297  std::vector<T> solution ( N );
298  for (ssize_t i = N - 1; i >= 0; --i) {
299  solution[i] = A.at(i, N) / A.at(i, i);
300  for (ssize_t k = i-1; k >= 0; --k) {
301  A.at(k, N) -= A.at(k, i) * solution[i];
302  }
303  }
304  return solution;
305 }
